## Column

Defines a single column in an Excel schema.

### Type Definition

```typescript
export type Column<
  T extends GenericObject,
  FieldValue extends string | ((data: T) => CellValue),
  ColKey extends string,
  TransformMap extends TransformersMap,
  FormatMap extends FormattersMap,
  Preset extends FormatterPreset<FormatMap>[keyof FormatMap] = never
> = {
  type: 'column'
  label?: string
  columnKey: ColKey
  key: FieldValue
  default?: CellValue
  format?: Preset | string | ((rowData: T, rowIndex: number, subRowIndex: number) => string | Preset)
  cellStyle?: CellStyle | ((rowData: T, rowIndex: number, subRowIndex: number) => CellStyle)
  headerStyle?: CellStyle
  summary?: Array<{
    value: (data: T[]) => BaseCellValue
    format?: string | Preset | ((data: T[]) => string | Preset)
    cellStyle?: CellStyle | ((data: T[]) => CellStyle)
  }>
} & (
  ExtractColumnValue<T, FieldValue> extends CellValue
    ? { transform?: TypedTransformersMap<TransformMap, ExtractColumnValue<T, FieldValue>> | ((value: ExtractColumnValue<T, FieldValue>, index: number) => CellValue) }
    : { transform: TypedTransformersMap<TransformMap, ExtractColumnValue<T, FieldValue>> | ((value: ExtractColumnValue<T, FieldValue>, index: number) => CellValue) }
)
```

### Type Parameters

<ParamField path="T" type="GenericObject">
  The data type for rows
</ParamField>

<ParamField path="FieldValue" type="string | ((data: T) => CellValue)">
  Either a property path string or a function to extract cell value
</ParamField>

<ParamField path="ColKey" type="string">
  Unique identifier for this column
</ParamField>

<ParamField path="TransformMap" type="TransformersMap">
  Available transformer functions
</ParamField>

<ParamField path="FormatMap" type="FormattersMap">
  Available formatter presets
</ParamField>

<ParamField path="Preset" type="FormatterPreset" default="never">
  Specific formatter preset type (inferred from usage)
</ParamField>

### Properties

<ResponseField name="type" type="'column'" required>
  Discriminator property (always `'column'`)
</ResponseField>

<ResponseField name="columnKey" type="string" required>
  Unique key for the column. Used for selection and identification.
</ResponseField>

<ResponseField name="key" type="string | function" required>
  Data accessor. Can be:
  - A property path string (e.g., `'customer.name'`, `'items.0.price'`)
  - A function that extracts the value: `(data: T) => CellValue`
</ResponseField>

<ResponseField name="label" type="string">
  Display label for the column header. Defaults to `columnKey` if not provided.
</ResponseField>

<ResponseField name="default" type="CellValue">
  Default value when the cell value is null or undefined.
</ResponseField>

<ResponseField name="transform" type="string | function">
  Value transformer. Can be:
  - Name of a registered transformer (type-checked)
  - Inline transformer function: `(value, index) => CellValue`
  
  **Required** if the extracted value type is not compatible with `CellValue`.
</ResponseField>

<ResponseField name="format" type="Preset | string | function">
  Cell number format. Can be:
  - A registered formatter preset with params: `{ preset: 'currency', params: { symbol: '$' } }`
  - An Excel format string: `'$#,##0.00'`
  - A dynamic function: `(rowData, rowIndex, subRowIndex) => string | Preset`
</ResponseField>

<ResponseField name="cellStyle" type="CellStyle | function">
  Cell style configuration. Can be:
  - A static `CellStyle` object
  - A dynamic function: `(rowData, rowIndex, subRowIndex) => CellStyle`
  
  Style properties include font, fill, border, alignment, etc. (from xlsx-js-style)
</ResponseField>

<ResponseField name="headerStyle" type="CellStyle">
  Style configuration for the column header cell.
</ResponseField>

<ResponseField name="summary" type="Array<SummaryRow>">
  Array of summary row configurations. Each summary row contains:
  - `value`: Function to compute summary value from all data
  - `format`: Optional format for the summary cell
  - `cellStyle`: Optional style for the summary cell
  
  Summary rows appear at the bottom of the table when `summary: true` is set.
</ResponseField>

### Examples

#### Basic Column

```typescript
schema.column('productName', {
  label: 'Product Name',
  key: 'name'
})
```

#### Nested Path Access

```typescript
schema.column('customerEmail', {
  label: 'Email',
  key: 'customer.contact.email'
})
```

#### Function-Based Value

```typescript
schema.column('itemCount', {
  label: '# Items',
  key: (order) => order.items.length
})
```

#### With Transformer

```typescript
schema
  .withTransformers({
    currency: (value: number) => `$${value.toFixed(2)}`
  })
  .column('price', {
    label: 'Price',
    key: 'amount',
    transform: 'currency' // Type-checked based on 'amount' type
  })
```

#### With Dynamic Styling

```typescript
schema.column('status', {
  label: 'Status',
  key: 'status',
  cellStyle: (order) => ({
    fill: { 
      fgColor: { 
        rgb: order.status === 'completed' ? '00FF00' : 'FFFF00' 
      } 
    },
    font: { 
      bold: order.status === 'urgent' 
    }
  })
})
```

#### With Summary

```typescript
schema.column('revenue', {
  label: 'Revenue',
  key: 'total',
  format: '$#,##0.00',
  summary: [
    {
      value: (data) => data.reduce((sum, row) => sum + row.total, 0),
      format: '$#,##0.00',
      cellStyle: {
        font: { bold: true, sz: 12 },
        fill: { fgColor: { rgb: 'D3D3D3' } }
      }
    },
    {
      value: (data) => data.reduce((sum, row) => sum + row.total, 0) / data.length,
      format: '$#,##0.00'
    }
  ]
})
```

## ColumnGroup

Defines a group of dynamically generated columns based on runtime context.

### Type Definition

```typescript
export interface ColumnGroup<
  T extends GenericObject,
  ColKey extends string,
  KeyPaths extends string,
  UsedKeys extends string,
  TransformMap extends TransformersMap,
  FormatMap extends FormattersMap,
  Context,
  ContextMap extends Record<string, any> = {}
> {
  type: 'group'
  columnKey: ColKey
  builder: () => ExcelSchemaBuilder<T, KeyPaths, UsedKeys, TransformMap, FormatMap>
  handler: GroupHandler<T, KeyPaths, UsedKeys, TransformMap, FormatMap, Context>
}
```

### Type Parameters

<ParamField path="T" type="GenericObject">
  The data type for rows
</ParamField>

<ParamField path="ColKey" type="string">
  Unique identifier for this column group
</ParamField>

<ParamField path="KeyPaths" type="string">
  Valid nested paths in the data type
</ParamField>

<ParamField path="UsedKeys" type="string">
  Already-used column keys
</ParamField>

<ParamField path="TransformMap" type="TransformersMap">
  Available transformer functions
</ParamField>

<ParamField path="FormatMap" type="FormattersMap">
  Available formatter presets
</ParamField>

<ParamField path="Context" type="any">
  Type of context data required by the handler
</ParamField>

<ParamField path="ContextMap" type="Record<string, any>" default="{}">
  Map of all context types in the schema
</ParamField>

### Properties

<ResponseField name="type" type="'group'" required>
  Discriminator property (always `'group'`)
</ResponseField>

<ResponseField name="columnKey" type="string" required>
  Unique key for the column group (must start with `'group:'`)
</ResponseField>

<ResponseField name="builder" type="function" required>
  Factory function that creates a builder instance with inherited transformers and formatters
</ResponseField>

<ResponseField name="handler" type="GroupHandler" required>
  Function that defines columns using the builder and runtime context
</ResponseField>

### GroupHandler Type

```typescript
export type GroupHandler<
  T extends GenericObject,
  CellKeyPaths extends string,
  UsedKeys extends string,
  TransformMap extends TransformersMap,
  FormatMap extends FormattersMap,
  Context
> = (
  builder: ExcelSchemaBuilder<T, CellKeyPaths, UsedKeys, TransformMap, FormatMap>,
  context: Context
) => void
```

### Example

```typescript
interface SalesData {
  product: string
  region: string
  sales: Record<string, number>
}

interface QuarterContext {
  quarters: Array<{ name: string; months: string[] }>
}

const schema = ExcelSchemaBuilder.create<SalesData>()
  .withFormatters({
    currency: '$#,##0.00'
  })
  .column('product', {
    label: 'Product',
    key: 'product'
  })
  .group<'group:quarters', QuarterContext>('group:quarters', (builder, context) => {
    context.quarters.forEach(quarter => {
      builder.column(quarter.name, {
        label: quarter.name,
        key: (data) => {
          return quarter.months.reduce((sum, month) => 
            sum + (data.sales[month] || 0), 0
          )
        },
        format: { preset: 'currency' },
        summary: [{
          value: (data) => data.reduce((sum, row) => {
            return sum + quarter.months.reduce((qSum, month) => 
              qSum + (row.sales[month] || 0), 0
            )
          }, 0)
        }]
      })
    })
  })
  .build()

// Usage:
ExcelBuilder.create()
  .sheet('sales')
  .addTable({
    schema,
    data: salesData,
    context: {
      'group:quarters': {
        quarters: [
          { name: 'Q1', months: ['Jan', 'Feb', 'Mar'] },
          { name: 'Q2', months: ['Apr', 'May', 'Jun'] }
        ]
      }
    }
  })
```

## CellValue

Represents valid values for Excel cells.

```typescript
export type BaseCellValue = string | number | boolean | null | undefined | Date
export type CellValue = BaseCellValue | BaseCellValue[]
```

Arrays of values create multiple sub-rows within a single row (cells are merged for other columns).
